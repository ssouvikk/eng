==========================================================================
VINC / VICT → conquer, overcome, defeat, win
==========================================================================

invincible → unconquerable; unbeatable
invincibility → unconquerable nature
conquer → overcome by force or victory
conqueror → one who conquers
conquerable → capable of being conquered
conquered → overcome by force
evince → reveal the presence of; show clearly
evident → clearly seen
victor → winner
victorious → triumphant
victory → triumph
victimize → make a victim of
victim → one who suffers
convince → persuade strongly
convincing → persuasive
conviction → firm belief
convict → someone found guilty of a crime
vanquish → conquer or defeat
victual → provision of food
victuals → provisions

==========================================================================


invincible : {
    in (not) + vinc (conquer) + ible (able)
    "not able to be conquered"
        → literal: impossible to defeat
        → actual: unconquerable; unbeatable
}

invincibility : {
    invincible + ity (state/quality)
    "state of being unconquerable"
        → literal: quality of being impossible to defeat
        → actual: unconquerable nature
}

conquer : {
    from Latin conquirere
    con (thoroughly) + querere (seek, obtain)
    "gain completely"
        → literal: obtain by overcoming resistance
        → actual: defeat; overcome by force
}

conqueror : {
    conquer + or (one who)
    "one who conquers"
        → literal: one who defeats others
        → actual: conqueror; winner
}

conquerable : {
    conquer + able (capable of)
    "capable of being conquered"
        → literal: able to be defeated
        → actual: capable of being overcome
}

conquered : {
    conquer + ed (past participle)
    "was conquered"
        → literal: was defeated
        → actual: overcome by force
}

evince : {
    e/ex (out) + vinc (conquer, prove)
    "bring out by proof"
        → literal: show clearly through evidence
        → actual: reveal; demonstrate
}

evident : {
    from Latin evidens
    e/ex (out) + vid (see)
    "seen clearly"
        → literal: plainly visible
        → actual: obvious; apparent
}

victor : {
    vict (conquer) + or (one who)
    "one who conquers"
        → literal: one who defeats others
        → actual: winner
}

victorious : {
    victor (winner) + ious (full of)
    "full of victory"
        → literal: characterized by winning
        → actual: triumphant
}

victory : {
    vict (conquer) + ory (state/result)
    "state of having conquered"
        → literal: result of winning
        → actual: triumph
}

victimize : {
    victim + ize (make)
    "make a victim"
        → literal: cause someone to suffer
        → actual: make a victim of
}

victim : {
    from Latin victima (sacrificial victim)
    "one sacrificed or harmed"
        → literal: one subjected to suffering
        → actual: sufferer
}

convince : {
    con (thoroughly) + vinc (conquer)
    "conquer thoroughly"
        → literal: overcome someone's doubts
        → actual: persuade strongly
}

convincing : {
    convince + ing (causing)
    "causing conviction"
        → literal: causing someone to be persuaded
        → actual: persuasive
}

conviction : {
    convince + tion (state/act)
    "state of being convinced"
        → literal: condition of being persuaded
        → actual: firm belief
}

convict : {
    con (thoroughly) + vict (prove, conquer)
    "prove thoroughly guilty"
        → literal: overcome with evidence
        → actual: find guilty; criminal
}

vanquish : {
    from Old French vainquiss-
    ultimately from Latin vincere (conquer)
    "conquer completely"
        → literal: defeat decisively
        → actual: vanquish; overcome
}

victual : {
    from Latin victus (food, nourishment)
    "that which sustains life"
        → literal: food supply
        → actual: provision of food
}

victuals : {
    victual + s
    "food provisions"
        → literal: supplies for nourishment
        → actual: provisions
}
