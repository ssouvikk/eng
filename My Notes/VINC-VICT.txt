==========================================================================
VINC / VICT → conquer, overcome, defeat, win
==========================================================================

invincible → unconquerable; unbeatable
invincibility → unconquerable nature
conquer → overcome by force or victory
conqueror → one who conquers
conquerable → capable of being conquered
conquered → overcome by force
evince → reveal the presence of; show clearly
evident → clearly seen
victor → winner
victorious → triumphant
victory → triumph
victimize → make a victim of
victim → one who suffers
convince → persuade strongly
convincing → persuasive
conviction → firm belief
convict → someone found guilty of a crime
vanquish → conquer or defeat
victual → provision of food
victuals → provisions

==========================================================================

invincible : {
    in (not) + vinc (conquer) + ible (able)
    semantic :
        → not able to be conquered
        → impossible to defeat
        → unconquerable; unbeatable
}

invincibility : {
    invincible + ity (state/quality)
    semantic :
        → state of not being conquerable
        → condition of being impossible to defeat
        → unconquerable nature
}

conquer : {
    con (thoroughly) + querere (seek, gain)
    semantic :
        → gain completely
        → gain by overcoming resistance
        → defeat and take control
    → overcome by force or victory
}

conqueror : {
    conquer + or (one who)
    semantic :
        → one who conquers
        → one who defeats opponents
        → winner; conqueror
}

conquerable : {
    conquer + able (capable of)
    semantic :
        → capable of being conquered
        → able to be defeated
        → capable of being overcome
}

conquered : {
    conquer + ed (past participle)
    semantic :
        → was conquered
        → was defeated
        → overcome by force
}

evince : {
    e/ex (out) + vinc (prove, overcome doubt)
    semantic :
        → bring out by proof
        → show clearly through evidence
        → reveal; demonstrate
}

evident : {
    e/ex (out) + vid (see)
    semantic :
        → seen clearly
        → plainly visible
        → obvious; apparent
}

victor : {
    vict (conquer) + or (one who)
    semantic :
        → one who conquers
        → one who defeats others
        → winner
}

victorious : {
    victor (winner) + ious (full of)
    semantic :
        → full of victory
        → characterized by success in battle or contest
        → triumphant
}

victory : {
    vict (conquer) + ory (state/result)
    semantic :
        → state of having conquered
        → successful defeat of an opponent
        → triumph
}

victimize : {
    victim + ize (make)
    semantic :
        → make into a victim
        → cause to suffer harm or injustice
        → make a victim of
}

victim : {
    Latin victima (sacrificial offering)
    semantic :
        → sacrificial being
        → one subjected to harm or loss
        → sufferer
}

convince : {
    con (thoroughly) + vinc (conquer)
    semantic :
        → conquer thoroughly
        → overcome resistance or doubt
        → persuade strongly
}

convincing : {
    convince + ing (causing)
    semantic :
        → causing someone to be convinced
        → overcoming doubt
        → persuasive
}

conviction : {
    convince + tion (state/act)
    semantic :
        → state of being convinced
        → settled belief after persuasion
        → firm belief
}

convict : {
    con (thoroughly) + vict (prove, conquer)
    semantic :
        → prove thoroughly guilty
        → establish guilt by evidence
        → find guilty of a crime
        → person found guilty of a crime
}

vanquish : {
    from Latin vincere (conquer)
    semantic :
        → conquer completely
        → defeat thoroughly
        → defeat decisively
}

victual : {
    from Latin victus (food, nourishment)
    semantic :
        → that which sustains life
        → food supply
        → provision of food
}

victuals : {
    victual + s
    semantic :
        → food supplies
        → provisions for sustenance
        → provisions
}
