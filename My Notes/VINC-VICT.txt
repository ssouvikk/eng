VINC-         → conquer, overcome, defeat
VICT-         → conquer, victory, winner

IN-           → not
CON-          → thoroughly, completely
E-/EX-        → out

-IBLE         → able to
-OR           → one who
-ION          → act, state, condition
-OUS          → full of
-IZE          → make
======================================================

INVINCIBLE : {
    in (not) + vinc (conquer) + ible (able)

    Semantic Evolution (literal → modern):
        → not able to be conquered
        → impossible to defeat
        → unconquerable; unbeatable
}

INVINCIBILITY : {
    invincible (unconquerable) + ity (state)

    Semantic Evolution (literal → modern):
        → state of being unconquerable
        → condition of resisting defeat
        → unbeatable nature
}
