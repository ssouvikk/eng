EQUI-          → equal, same
PAR-           → equal, equivalent

VOC- / VOX-    → voice, call

LATER-         → side
EQU-           → horse

NOCT-          → night

ANIM-          → mind, spirit
LIBRA-         → balance, scale

DIS-           → apart, away, not
IN-            → not

-AL            → relating to
-ITY           → state, quality
-IST           → person
======================================================

EQUIVOCAL        → having more than one meaning
VOCIFEROUS       → loud; noisy
VOCALIZE         → express in words or voice
VOCALIST         → singer

EQUILATERAL      → having equal sides

EQUIVOCAL : {
    equi (equal) + vocal (voice)

    Semantic Evolution (literal → modern):
        → equal voice in two directions
        → capable of multiple interpretations
        → having more than one meaning
}

VOCIFEROUS : {
    voc (voice) + ferous (carrying)

    Semantic Evolution (literal → modern):
        → carrying the voice
        → loudly expressive
        → noisy and outspoken
}

VOCALIZE : {
    vocal (voice) + ize (make)

    Semantic Evolution (literal → modern):
        → put into voice
        → express aloud
        → utter in spoken or sung form
}

VOCALIST : {
    vocal (voice) + ist (person)

    Semantic Evolution (literal → modern):
        → person using the voice
        → singer
        → professional vocal performer
}

EQUILATERAL : {
    equi (equal) + lateral (side)

    Semantic Evolution (literal → modern):
        → equal-sided
        → having sides of the same length
        → geometrically equal on all sides
}

EQUESTRIAN : {
