=================================================================
GEN → birth, produce, kind (generate, genesis, genetics)
GENIT → birth, beget (genital, congenital, progenitor)
GNAT / GNOS → born, known by birth/kinship (agnate, cognate)
=================================================================

gene → unit of inheritance
genus → kind; class; category
genesis → beginning, origin
generate → produce
generator → one who produces
generation → production; produced group
generational → relating to generations
generations → groups of people born during the same period
generation gap → gap between generations

genetic → relating to inheritance
genetics → study of inheritance
geneticist → one who studies inheritance

genealogy → study of lineage
genealogist → one who studies lineage
genealogical → relating to lineage

progenitor → ancestor
progeny → offspring
progenerate → produce offspring

congenital → existing from birth
genital → relating to reproduction
genitourinary → relating to reproduction and urinary organs
genitive → relating to origin or birth

indigenous → native; born in a place
aboriginal → existing from the beginning; native
autochthonous → indigenous; native to a region

genuine → authentic; true to its original nature

regenerate → renew; restore; produce again
regeneration → renewal; restoration

degenerate → become worse; decay
degeneration → decline; deterioration

homogeneous → of the same kind
heterogeneous → of different kinds

genius → exceptional natural ability
ingenious → clever; inventive
congenial → pleasant because naturally suited
genial → cheerful and friendly
geniality → cheerfulness and friendliness
genteel → refined; polite and well-bred

indignation → resentment at unfair treatment

agnate → relative through the male line
agnatic → relating to male-line descent
cognate → related by common origin
cognation → blood relationship; kinship

gene : {
    gene (birth, origin, production)
    "birth/origin"
        → literal: unit of origin or inheritance
        → actual: basic unit of heredity
},

genus : {
    gen (birth, race, kind)
    "kind/race"
        → literal: a kind or class produced from the same stock
        → actual: a biological category above species
},

genesis : {
    gen (birth/origin) + sis (state/process/result)
    "birth" + "state/process"
        → literal: process of coming into being
        → actual: origin; beginning; creation
},

generate : {
    gener (produce, beget) + ate (make/do)
    "produce" + "make"
        → literal: bring forth
        → actual: create; produce; cause to exist
},

generator : {
    generate + or (one who does)
    "produce" + "one who"
        → literal: one that produces
        → actual: a device or person that generates
},

generation : {
    gener (produce, beget) + ation (act/process/result)
    "produce" + "process"
        → literal: act of producing
        → actual: production; people born in the same period
},

genetic : {
    genet (origin, production) + ic (relating to)
    "origin" + "relating to"
        → literal: relating to origin
        → actual: relating to genes and heredity
},

genetics : {
    genetic + s
    "inheritance"
        → literal: science of inheritance
        → actual: study of genes and heredity
},

geneticist : {
    genetic + ist (specialist)
    "inheritance" + "specialist"
        → literal: specialist in genetics
        → actual: one who studies heredity
},

    gene (birth/origin) + a (ancestry/lineage)
    "birth/origin" + "ancestry"
        → literal: lineage traced from an origin
        → actual: family descent
},

genealogy : {
    genea (ancestry) + logy (study of)
    "ancestry" + "study"
        → literal: study of ancestry
        → actual: study of family history and descent
},

genealogist : {
    genealogy + ist (specialist)
    "study of ancestry" + "specialist"
        → literal: specialist in genealogy
        → actual: one who studies family lineage
},

genealogical : {
    genealogy + ical (relating to)
    "genealogy" + "relating to"
        → literal: relating to ancestry
        → actual: concerning family lineage
},

progenitor : {
    pro (forward, before) + genit (beget) + or (one who)
    "beget beforehand"
        → literal: one who came before and begot
        → actual: ancestor
},

    "beget forth"

        → actual: offspring; descendants

progenerate : {

    "forth" + "produce"

        → actual: reproduce; beget

genitive : {

        → actual: grammatical case showing possession
