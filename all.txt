EGO ("I" or "self") {

    Egoist : {
        = ego + ist
        = self-interested person
        = person concerned with own welfare
    },

    Egotist : {
        = ego + tist          // T = Talks
        = person who constantly talks about self
        = boastful self-promoter
    },

    Egocentric : {
        = ego + centrum
        = self at the center
        → self-centered person
    },

    Egomaniac : {
        = ego + mania
        = obsession/madness with self
        → pathological self-importance
    },

    Egoism : {
        = ego + ism
        = doctrine of self-interest
        → selfish philosophy
    },

    Egotism : {
        = ego + ism
        = excessive self-importance
        → vanity and boastfulness
    },
}


ALTER ("other") {

    Alter Ego : {
        = alter + ego
        = other self
        = second personality
    },

    Alternate : {
        = alter + ate
        = become the other
        → take turns 
        → substitute successively
    },

    Alternative : {
        = alter + native
        = other choice
        → another possibility
        → option
    },

    Alteration : {
        = alter + ation
        = act of making other
        → modification
        → change
    },

    Altercation : {
        = alter + cation
        = becoming different in opinion
        → disagreement
        → argument
        → quarrel
    },
}


ALTRUIST ("other people first") {

    Altruist : {
        = alter → altrui + ist
        = person devoted to others
        = selfless helper
    },

    Altruism : {
        = alter → altrui + ism
        = philosophy of helping others
        → selfless concern for others
    },

    Altruistic : {
        = alter → altrui + istic
        = characterized by selflessness
    },
}


CENTRUM ("center") {

    Egocentric : {
        = ego + centrum
        = self at the center
        → self-centered person
    },
}


MANIA ("madness" or "obsession") {

    Egomaniac : {
        = ego + mania
        = madness about self
        → obsessive self-importance
    },
}
