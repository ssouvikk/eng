GERIATRICS : {
    geri (old age) + iatrics (medical treatment)

    Semantic Evolution (literal → modern):
        → treatment of old age
        → medical care for elderly people
        → branch of medicine dealing with aging and the aged
}

GERIATRICIAN : {
    geri (old age) + iatrician (medical specialist)

    Semantic Evolution (literal → modern):
        → physician for the aged
        → specialist in elderly care
        → doctor trained in geriatric medicine
}

    odont (tooth) + algia (pain)

    Semantic Evolution (literal → modern):
        → tooth pain
        → pain arising from a tooth
        → toothache
}

NOSTALGIA : {
    nostos (return home) + algia (pain, longing)

    Semantic Evolution (literal → modern):
        → pain from wishing to return home

        → sentimental longing for the past
}

CARDIALGIA : {
    cardi (heart) + algia (pain)

    Semantic Evolution (literal → modern):
        → heart pain

}

INTERNIST : {
    intern (internal) + ist (specialist)

    Semantic Evolution (literal → modern):
        → specialist of internal conditions

        → doctor treating adult diseases without surgery
}

    Semantic Evolution (literal → modern):
