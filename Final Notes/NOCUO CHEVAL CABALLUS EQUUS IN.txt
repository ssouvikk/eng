INNOCUOUS : {
    in (not) + nocuo (harm)

    Semantic Evolution (literal → modern):
        → not harmful
        → causing no injury
        → harmless and unlikely to offend or damage
}

INNOCENT : {
    in (not) + nocens (harming)

    Semantic Evolution (literal → modern):
        → not harming
        → free from guilt
        → harmless, pure, or blameless
}

NOXIOUS : {
    nocuo (harm)

    Semantic Evolution (literal → modern):
        → harmful
        → injurious to health
        → poisonous, dangerous, or destructive
}

NOCUOUS : {
    nocuo (harm)

    Semantic Evolution (literal → modern):
        → harmful
        → damaging
        → capable of causing injury or harm
}

INNOCUOUSNESS : {
    in (not) + nocuo (harm)

    Semantic Evolution (literal → modern):
        → absence of harm
        → harmless quality
        → state of being safe and inoffensive
}
