GLIB : {
    origin uncertain (possibly related to smoothness)

    Semantic Evolution (literal → modern):
        → smooth in speech
        → speaking easily without depth
        → fluent but superficial
}

GLIBLY : {
    glib (superficially fluent) + ly (manner)

    Semantic Evolution (literal → modern):
        → in a smooth manner
        → with effortless fluency
        → superficially or insincerely
}

GLIBNESS : {
    glib (superficially fluent) + ness (quality)

    Semantic Evolution (literal → modern):
        → quality of smooth speech
        → readiness of expression
        → superficial eloquence
}

GREGARIOUS : {
    greg (herd, flock) + arious (belonging to)

    Semantic Evolution (literal → modern):
        → belonging to a herd
        → preferring company of others
        → sociable and outgoing
}

GREGARIOUSNESS : {
    gregarious (sociable) + ness (quality)

    Semantic Evolution (literal → modern):
        → tendency to join groups
        → sociability
        → enjoyment of companionship
}

EGREGIOUS : {
    e (out of) + greg (herd)

    Semantic Evolution (literal → modern):
        → standing out from the herd
        → remarkably noticeable
        → outstandingly bad or shocking
}

EGREGIOUSNESS : {
    egregious (outrageous) + ness (quality)

    Semantic Evolution (literal → modern):
        → quality of standing out
        → remarkable offensiveness
        → extreme badness
}

    con (together) + greg (herd)

        → gather into a flock

        → gathering together

        → bring into a group

CONSCIENCE : {

    Semantic Evolution (literal → modern):
