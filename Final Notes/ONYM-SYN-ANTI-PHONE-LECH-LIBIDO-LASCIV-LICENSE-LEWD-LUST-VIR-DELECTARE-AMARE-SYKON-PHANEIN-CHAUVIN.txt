SYNONYM : {
    syn (same) + onym (name)

    Semantic Evolution (literal → modern):
        → same name
        → different words with the same meaning
        → word equivalent in meaning to another
}

ANTONYM : {
    anti (opposite) + onym (name)

    Semantic Evolution (literal → modern):
        → opposite name
        → word with opposite meaning
        → linguistic opposite of another word
}

HOMONYM : {
    homo (same) + onym (name)

    Semantic Evolution (literal → modern):
        → same name
        → words sharing form
        → words identical in spelling or pronunciation but different in meaning
}

HOMOPHONE : {
    homo (same) + phone (sound)

    Semantic Evolution (literal → modern):
        → same sound
        → words pronounced alike
        → words with identical pronunciation but different meanings
}
