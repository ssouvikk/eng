ICONOCLASM : {
    eikon (image) + clasm (breaking)

    Semantic Evolution (literal → modern):
        → breaking images
        → destruction of religious symbols
        → rejection of traditions and established beliefs
}

PRURIENT : {
    pruri (itch)

    Semantic Evolution (literal → modern):
        → itching
        → craving intensely
        → characterized by excessive sexual curiosity or desire
}

    pruri (itch) + itus (condition)

    Semantic Evolution (literal → modern):
        → itching condition
        → persistent skin irritation
        → medical term for itching
}
