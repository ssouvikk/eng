UTERUS : {
    Latin uterus (womb)

    Semantic Evolution (literal → modern):
        → womb
        → female reproductive organ
        → organ where the fetus develops during pregnancy
}

UTERINE : {
    uter (womb) + ine (relating to)

    Semantic Evolution (literal → modern):
        → relating to the womb
        → pertaining to the uterus
        → associated with uterine structures or functions
}

UTERECTOMY : {
    uter (womb) + ectomy (surgical removal)

    Semantic Evolution (literal → modern):
        → removal of the womb
        → surgical excision of the uterus
        → medical operation removing the uterus
}

INTRAUTERINE : {
    intra (within) + uter (womb) + ine (relating to)

    Semantic Evolution (literal → modern):
        → within the womb
        → occurring inside the uterus
        → situated in the uterine cavity
}

    extra (outside) + uter (womb) + ine (relating to)

        → outside the womb

}

HYSTERIA : {
    hystera (womb)

    Historical Context:
        → ancient belief that a wandering uterus caused symptoms
        → associated with emotional and physical disturbances

    Semantic Evolution (literal → modern):
        → condition attributed to the womb
        → excessive emotional excitement
        → uncontrollable emotional outburst
}

    hysteria + al (relating to)

    Semantic Evolution (literal → modern):
