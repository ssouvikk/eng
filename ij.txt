IATREIA ("healing", "medical treatment") {

    Geriatrics : {
        = geras + iatreia
        = healing of old age
        → medical care of the elderly
        → geriatrics
    },

    Pediatrician : {
        = paidos + iatreia + ian
        = children's physician
        → pediatrician
    },

    Pediatrics : {
        = paidos + iatreia
        = healing of children
        → medical care of children
    },

    Podiatrist : {
        = podos + iatreia + ist
        = foot healer
        → foot doctor
        → podiatrist
    },

    Podiatry : {
        = podos + iatreia
        = healing of the foot
        → treatment of foot ailments
        → podiatry
    },

    Psychiatrist : {
        = psyche + iatreia + ist
        = physician healing the mind
        → psychiatrist
    },

    Psychiatry : {
        = psyche + iatreia
        = healing of the mind
        → treatment of mental disorders
        → psychiatry
    },
}


ICON ("image") {

    Iconoclasm : {
        = icon + clasm
        = destruction of images
        → hostility to tradition
        → rejection of established beliefs
    },

    Iconoclast : {
        = icon + klastes
        = breaker of images
        → attacker of cherished beliefs
        → scoffer at tradition
        → rebel against convention
    },

    Iconoclastic : {
        = iconoclast + ic
        = attacking accepted beliefs
        → unconventional
        → anti-traditional
    },
}


INTERNUS ("inside") {

    Intern : {
        = internus
        = one working inside
        → medical apprentice in a hospital
        → intern
    },

    Internal : {
        = internus
        = inside
        → located within
        → internal
    },

    Internist : {
        = internus + ist
        = specialist of the inside
        → physician studying internal organs
        → expert in internal medicine
    },
}
